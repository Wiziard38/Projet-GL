package fr.ensimag.deca.tree;

/**
 * List of expressions (eg list of parameters).
 *
 * @author gl39
 * @date 01/01/2023
 */
public class ListExpr extends TreeList<AbstractExpr> {

    @Override
    public void decompile(IndentPrintStream s) {
        for (AbstractExpr c : getList()) {
            c.decompile(s);
            s.println();
        }
    }
